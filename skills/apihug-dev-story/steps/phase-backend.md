  <!-- ApiHug Immutable Constraints (ZERO TOLERANCE) -->
  <!-- ============================================ -->

  <!-- Architecture -->
  <critical>Onion Architecture: Adapter (ServiceImpl) -> Domain (DomainService) -> Infrastructure (Repository)</critical>
  <critical>Source Set: NEVER modify src/generated/</critical>
  <critical>Generated Code — FORBIDDEN to remove class-level or method-level comments in ServiceImpl/Repository/Controller/Entity/DAO</critical>
  <critical>Lombok: FORBIDDEN — write explicit getters/setters/constructors</critical>

  <!-- Proto Core Constraints -->
  <critical>Proto DSL: snake_case for fields; API MUST NOT import domain entities</critical>
  <critical>NEVER add page/size/sort to Request Message — use pageable: true</critical>
  <critical>NEVER wrap response in PageableXxx — framework wraps Result&lt;Pageable&lt;T&gt;&gt;</critical>
  <critical>NEVER use string for enum values — use EnumType with full package path</critical>
  <critical>NEVER add file-level options (java_package, go_package) — build system injects</critical>
  <critical>NEVER import domain/ entities in api/ layer — API layer must NOT depend on Domain</critical>
  <critical>NEVER use SQL reserved words as column names — use prefixes: user_name, order_no</critical>
  <critical>NEVER use relative type references — always use full package: com.example.Order order = 1</critical>
  <critical>NEVER create Request Message with only 1-2 fields — use google.protobuf.Empty + parameters{}</critical>
  <critical>NEVER create proto message with single repeated field — use input_repeated/output_repeated</critical>
  <critical>NEVER use DTO in message names — MUST have real business meaning</critical>
  <critical>NEVER put all code in root api/, domain/, infra/ — Each feature gets sub dir</critical>
  <critical>NEVER define INNER message type — all messages at file level</critical>
  <critical>NEVER add redundant ENUM suffix — STATUS_ACTIVE yes, STATUS_ENUM_ACTIVE no</critical>
  <critical>NEVER use string with enum values in comments</critical>
  <critical>NEVER modify _full.xml migration file — auto-generated by wire</critical>
  <critical>NEVER use ORDINAL for enum persistence — use STRING or CODE</critical>
  <critical>Table names MUST have prefix: SYS_, BIZ_, CFG_, LOG_ — avoid reserved words</critical>
  <critical>NEVER duplicate code values within same enum</critical>
  <critical>NEVER use numeric values for enum fields — use enum names: http_status: NOT_FOUND</critical>
  <critical>NEVER add any duplicate inline comment if the description is already sufficient; maintain code clarity and conciseness.</critical>
  <critical>NEVER define duplicate API path with a prefix that duplicates {apiRoot} if it is already defined</critical>


  <!-- Implementation Core Constraints -->
  <critical>FORBIDDEN remove existing class-level documentation; appending new content is permitted, but the original must remain intact.</critical>
  <critical>NEVER paginate in memory (subList/skip/limit on full result) — ALWAYS use DB-level pagination</critical>
  <critical>ServiceImpl is AUTO-GENERATED by wire — FORBIDDEN to create manually. Only add/implement method bodies.</critical>
  <critical>NEVER write query logic in ServiceImpl — ALL queries in Repository trait</critical>
  <critical>NEVER call builder.done() — controller handles response</critical>
  <critical>NEVER throw RuntimeException for business errors — use HopeErrorDetailException</critical>
  <critical>NEVER create class in the {trait_path}</critical>

  <!-- Positive Constraints -->
  <critical>ALWAYS run gradlew :{module}:wire after proto changes</critical>
  <critical>ALWAYS use snake_case for proto fields</critical>
  <critical>ALWAYS define enums in infra/{feature}_constant.proto</critical>
  <critical>ALWAYS use full package path for types: com.example.user.infra.UserStatusEnum</critical>
  <critical>ALWAYS define errors with code, http_status, phase, severity</critical>
  <critical>ALWAYS put query logic in Repository trait</critical>
  <critical>ALWAYS use HopeContextHolder.customer() for current user</critical>
  <critical>ALWAYS add `date_format` for date/datetime/time kind fields explicitly</critical>

  <!-- Execution Constraints -->
  <critical>Execute ALL steps in exact order; do NOT skip steps</critical>
  <critical>Absolutely DO NOT stop because of "milestones", "significant progress", or "session boundaries"</critical>
  <critical>FOLLOW THE STORY FILE TASKS/SUBTASKS SEQUENCE EXACTLY AS WRITTEN - NO DEVIATION</critical>

  <critical>NEVER proceed to next task until current task/subtask is complete AND tests pass</critical>
  <critical>Execute continuously without pausing until all tasks/subtasks are complete or explicit HALT condition</critical>

    <action>Determine task type: "proto"/"API design"/"entity"/"enum"/"error" -> PROTO_PHASE; "implement"/"service"/"repository" -> IMPL_PHASE</action>

    <!-- ==================== -->
    <!-- PROTO DESIGN PHASE   -->
    <!-- L2: Conditional deep loading based on task scope -->
    <!-- ==================== -->
    <check if="task type == PROTO_PHASE">

      <check if="task involves API design (service/rpc/request/response)">
        <action>Load {apihug_proto_api_extension_guide} — API/Service/RPC design, OpenAPI/Swagger annotations</action>
        <critical>PRE-WRITE CHECK: Pageable API? -> Check Pageable API Design Rules</critical>
        <critical>PRE-WRITE CHECK: Request Message? -> Check Parameter Definition rules</critical>
        <critical>PRE-WRITE CHECK: Query naming? -> SearchXxx for paginated, ListXxx for non-paginated, GetXxx for single</critical>
      </check>

      <check if="task involves Database design (entity/table/column)">
        <action>Load {apihug_proto_database_modeling_guide} — Entity/Table/Column definitions, persistence annotations</action>
        <critical>PRE-WRITE CHECK: Entity/Table? -> Check SQL Reserved Words table</critical>
        <critical>PRE-WRITE CHECK: Column definitions? -> Check Type and Length conventions</critical>
      </check>

      <check if="task involves Enum/Error design (constant/error code)">

        <critical>PRE-WRITE CHECK: Error codes? -> Check Error Sub-Message Fields</critical>
      </check>

      <critical>HALT if ANY MANDATORY rule violated — See loaded rules file MANDATORY EXECUTION RULES section</critical>

      <critical>PRE-WRITE CHECK: Pagination? -> Check rules RESULT and PAGEABLE section</critical>

      <action>Implement organized by FEATURE:
        - ServiceImpl: {java_path}/api/{Service}ServiceImpl.java (flat, wire-generated)
